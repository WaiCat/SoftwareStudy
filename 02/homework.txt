			<li>
				<label for="second">두번째 수</label>
				<input type="text" id="second">
			</li>
			<li>
				<button onclick="showPrice('1');">+</button>
                <button name="op" value="2" onclick="showPrice('2')">-</button>
                <button name="op" value="3" onclick="showPrice('3')">*</button>
                <button name="op" value="4" onclick="showPrice('4')">/</button>
                <button name="op" value="5" onclick="showPrice('5')">%</button>
                <button name="op" value="6"onclick="showPrice('6')">clear</button>
			</li>
		</ul>
		<div id="showResult"></div>	
	</div>	
		
	<script>
		function showPrice(op) {
			var frist = document.querySelector('#frist').value;  // 첫번째 수
			var second = document.querySelector('#second').value; // 두번째 수
            var calculation;
            var result;
            
			if(op=='1'){
                calculation = "합은 ";
                result = parseInt(frist) + parseInt(second);
            }else if(op=='2'){
                calculation = "차는 ";
                result = parseInt(frist) - parseInt(second);
            }else if(op=='3'){
                calculation = "곱은 ";
                result = parseInt(frist) * parseInt(second);
            }else if(op=='4'){
                calculation = "나눈값 ";
                result = parseInt(frist) / parseInt(second);
            }else if(op=='5'){
                calculation = "나머지는 ";
                result = parseInt(frist) % parseInt(second);
            }

			if(op=='6'){
				document.getElementById("frist").value ='';
				document.getElementById("second").value ='';
                document.querySelector('#showResult').innerHTML = "null";
            }else{
				document.querySelector('#showResult').innerHTML = " 두수의 " + calculation + result + "입니다";
			}
		}		
	</script>
